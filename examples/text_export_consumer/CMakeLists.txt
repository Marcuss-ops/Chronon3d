# ═══════════════════════════════════════════════════════════════════════════
# Chronon3D — Text Export Consumer Example
#
# Minimal standalone project that demonstrates how to use Chronon3D's
# public API to render text via SceneBuilder + l.text().
#
# Prerequisites:
#   Chronon3D installed via cmake --install (see README.md for steps).
#
# Build:
#   mkdir build && cd build
#   cmake .. -DCMAKE_PREFIX_PATH=/path/to/chronon3d-install
#   cmake --build .
#
# Run:
#   ./text_export_consumer
#   # → produces text_export_output.png in CWD
# ═══════════════════════════════════════════════════════════════════════════

cmake_minimum_required(VERSION 3.27)
project(text_export_consumer VERSION 1.0.0 LANGUAGES CXX)

# ── Find the installed Chronon3D package ──────────────────────────────────
# The CMAKE_PREFIX_PATH must point to the install prefix containing
# Chronon3DConfig.cmake (generated by cmake --install).
find_package(Chronon3D CONFIG REQUIRED)

# Guard: the documented public API target must be intact.
if(NOT TARGET Chronon3D::SDK)
    message(FATAL_ERROR
        "Chronon3D::SDK target not found. "
        "Ensure CMAKE_PREFIX_PATH points to a valid Chronon3D install prefix.")
endif()

# ── Build the consumer ───────────────────────────────────────────────────
add_executable(text_export_consumer main.cpp)

set_target_properties(text_export_consumer PROPERTIES
    CXX_STANDARD 20
    CXX_STANDARD_REQUIRED ON
)

# Link ONLY the public SDK target — no internal headers needed.
target_link_libraries(text_export_consumer PRIVATE Chronon3D::SDK)
